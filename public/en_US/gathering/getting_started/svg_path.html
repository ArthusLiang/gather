JAVASCRIPT

var motion = mframe([{
  dom: document.getElementById('momentum-logo'),
  frames: [
    { attr: { 'stroke-dashoffset': '600' }, time: 0 },
    { attr: { 'stroke-dashoffset': '0' }, time: 120, tween: 'easeInOut' },
    { attr: { 'stroke-dashoffset': '0' }, time: 150 },
  ]
}]);
motion.repeat(Infinity);
